<body>
    <div class="container">
        <h1>Will you be my Valentine? ❤️</h1>
        <div class="buttons">
            <button id="yes" onclick="sayYes()">Yes</button>
            <button id="no" onmouseover="moveNo()" onclick="swapButtons()">No</button>
        </div>
        <p id="message"></p>
    </div>

    <script>
        function sayYes() {
            document.getElementById("message").innerHTML = "Yay! ❤️ You made my day!";
        }

        function moveNo() {
            let x = Math.random() * window.innerWidth * 0.8;
            let y = Math.random() * window.innerHeight * 0.8;
            document.getElementById("no").style.left = `${x}px`;
            document.getElementById("no").style.top = `${y}px`;
        }

        function swapButtons() {
            let yesBtn = document.getElementById("yes");
            let noBtn = document.getElementById("no");

            // Swapping text and functionality
            yesBtn.innerText = "No";
            noBtn.innerText = "Yes";

            yesBtn.id = "no";
            noBtn.id = "yes";

            yesBtn.setAttribute("onmouseover", "moveNo()");
            noBtn.setAttribute("onclick", "sayYes()");
        }
    </script>
